# 研究下umi的结构, 研究两个版本 umi3/umi4
# 使用脚手架创建umi-app, 这个是umi3
  mkdir umi3-app
  cd umi3-app
  pnpm add --save-dev @umijs/create-umi-app
  npx @umijs/create-umi-app  --> 创建项目
  pnpm i                     --> 安装依赖
  pnpm start                 --> 测试
  



# 也可以使用ant-design/pro-cli 脚手架来创建项目
pnpm dlx @ant-design/pro-cli create ant-design-app
-> 选择 umi4/umi3
   这里选择的umi4
